112  template <class T>
113  void addRotated(const StateTwo &state, const size_t &idx,
114  std::vector<Eigen::Triplet<T>> &triplets, WignerD &wigner, const double &alpha,
115  const double &beta, const double &gamma) {
116  // Check whether the angles are compatible to the used data type
117  double tolerance = 1e-16;
118  if (std::is_same<T, double>::value &&
119  std::abs(std::remainder(alpha, 2 * M_PI)) > tolerance) {
120  throw std::runtime_error(
121  "If the Euler angle alpha is not a multiple of 2 pi, the Wigner D matrix element "
122  "is complex and cannot be converted to double.");
123  }
124  if (std::is_same<T, double>::value &&
125  std::abs(std::remainder(gamma, 2 * M_PI)) > tolerance) {
126  throw std::runtime_error(
127  "If the Euler angle gamma is not a multiple of 2 pi, the Wigner D matrix element "
128  "is complex and cannot be converted to double.");
129  }
130 
131  // Add rotated triplet entries
132  std::vector<T> val2_vector;
133  val2_vector.reserve(2 * state.getSecondState().getJ() + 1);
134 
135  for (float m2 = -state.getSecondState().getJ(); m2 <= state.getSecondState().getJ(); ++m2) {
136  val2_vector.push_back(
137  utils::convert<T>(wigner(state.getSecondState().getJ(), m2,
138  state.getSecondState().getM(), -gamma, -beta, -alpha)));
139  }
140 
141  for (float m1 = -state.getFirstState().getJ(); m1 <= state.getFirstState().getJ(); ++m1) {
142  auto val1 =
143  utils::convert<T>(wigner(state.getFirstState().getJ(), m1,
144  state.getFirstState().getM(), -gamma, -beta, -alpha));
145 
146  for (float m2 = -state.getSecondState().getJ(); m2 <= state.getSecondState().getJ();
147  ++m2) {
148  StateTwo newstate(state.getSpecies(), state.getN(), state.getL(), state.getJ(),
149  {{m1, m2}});
150  auto state_iter = states.get<1>().find(newstate);
151 
152  if (state_iter != states.get<1>().end()) {
153  auto val = val1 * val2_vector[m2 + state.getSecondState().getJ()];
154  triplets.push_back(Eigen::Triplet<T>(state_iter->idx, idx, val));
155  } else {
156  std::cerr << "Warning: Incomplete rotation because the basis is lacking some "
157  "Zeeman levels."
158  << std::endl;
159  }
160  }
161  }
162  }
