28 void GreenTensor::addSurface(double d) {
29  if (y != 0) {
30  throw std::runtime_error("The atoms must be in the xz-plane if a surface is present");
31  }
32 
33  double angle = std::atan(x / z);
34 
35  zA = d - z * std::sin(angle) / 2.;
36  zB = d + z * std::sin(angle) / 2.;
37 
38  if (zA < 0 || zB < 0) {
39  throw std::runtime_error(
40  "zA or zB < 0. One of the atoms is inside the plate. Plate is half-space z < 0.");
41  }

82 eigen_matrix33 GreenTensor::getDDTensorVacuum(double x, double y, double z) const {
83  // Build distance vector
84  Eigen::Matrix<double, 3, 1> distance;
85  distance << x, y, z;
86 
87  // Construct Green tensor
88  eigen_matrix33 vacuum_tensor =
89  -Eigen::MatrixXd::Identity(3, 3) / std::pow(distance.norm(), 3.) +
90  3. * distance * distance.transpose() / std::pow(distance.norm(), 5.);
91 
92  return vacuum_tensor;
93 }

95 eigen_matrix33 GreenTensor::getDDTensorPlate(double x, double zA, double zB) const {
96  // Calculate distances to mirror dipole
97  double zp = zA + zB;
98  double rp = std::sqrt(x * x + zp * zp);
99 
100  // Construct Green tensor
101  eigen_matrix33 plate_tensor_second_matrix;
102  plate_tensor_second_matrix << x * x, 0., -x * zp, 0., 0., 0., x * zp, 0., x * x;
103 
104  eigen_matrix33 plate_tensor =
105  Eigen::Vector3d({1., 1., 2.}).asDiagonal().toDenseMatrix() / std::pow(rp, 3.) -
106  3. * plate_tensor_second_matrix / std::pow(rp, 5);
107 
108  return plate_tensor;
109 }

111 eigen_tensor333 GreenTensor::getDQTensorVacuum(double x, double y, double z) const {
112  eigen_tensor333 vacuum_tensor;
113  vacuum_tensor.setZero();
114 
115  eigen_matrix33 Eye = eigen_matrix33::Identity(3, 3);
116  Eigen::Matrix<double, 3, 1> distance;
117  distance << x, y, z;
118  double dist = sqrt(x * x + y * y + z * z);
119 
120  for (int i = 0; i < 3; i++) {
121  for (int j = 0; j < 3; j++) {
122  for (int k = 0; k < 3; k++) {
123  vacuum_tensor(i, j, k) += -3 / std::pow(dist, 5.) *
124  (Eye(i, j) * distance(k) + distance(i) * Eye(j, k) +
125  distance(j) * Eye(i, k)) +
126  15. / std::pow(dist, 7.) * distance(i) * distance(j) * distance(k);
127  }
128  }
129  }
130 
131  return vacuum_tensor;
132 }

158 eigen_tensor333 GreenTensor::getDQTensorPlate(double x, double zA, double zB) const {
159  eigen_tensor333 plate_tensor;
160  plate_tensor.setZero();
161 
162  double zp = zA + zB;
163  Eigen::Matrix<double, 3, 1> distanceplus;
164  distanceplus << -x, 0., zp;
165  double rp = distanceplus.norm();
166 
167  eigen_matrix33 plate_tensor_first_matrix =
168  Eigen::Vector3d({1., 1., 2.}).asDiagonal().toDenseMatrix();
169  eigen_matrix33 plate_tensor_second_matrix;
170  plate_tensor_second_matrix << x * x, 0., -x * zp, 0., 0., 0., x * zp, 0., x * x;
171 
172  eigen_tensor333 plate_tensor_second_gradient;
173  plate_tensor_second_gradient.setZero();
174  plate_tensor_second_gradient(0, 0, 0) = -2. * x;
175  plate_tensor_second_gradient(2, 2, 0) = -2. * x;
176  plate_tensor_second_gradient(0, 2, 0) = zp;
177  plate_tensor_second_gradient(2, 0, 0) = -zp;
178  plate_tensor_second_gradient(0, 2, 2) = -x;
179  plate_tensor_second_gradient(2, 0, 2) = x;
180 
181  for (int i = 0; i < 3; i++) {
182  for (int j = 0; j < 3; j++) {
183  for (int k = 0; k < 3; k++) {
184  plate_tensor(i, j, k) +=
185  -3. * plate_tensor_first_matrix(i, j) * distanceplus(k) / std::pow(rp, 5.);
186  plate_tensor(i, j, k) +=
187  15. * plate_tensor_second_matrix(i, j) * distanceplus(k) / std::pow(rp, 7.);
188  plate_tensor(i, j, k) +=
189  -3. * plate_tensor_second_gradient(i, j, k) / std::pow(rp, 5.);
190  }
191  }
192  }
193 
194  return plate_tensor;
195 }
