52 void SystemOne::setBfield(std::array<double, 3> field) {
53  this->onParameterChange();
54  bfield = field;
55 
56  // Transform the magnetic field into spherical coordinates
57  this->changeToSphericalbasis(bfield, bfield_spherical);
58 
59  diamagnetism_terms[{{0, +0}}] = bfield_spherical[+0] * bfield_spherical[+0] -
60  bfield_spherical[+1] * bfield_spherical[-1] * 2.;
61  diamagnetism_terms[{{2, +0}}] =
62  bfield_spherical[+0] * bfield_spherical[+0] + bfield_spherical[+1] * bfield_spherical[-1];
63  diamagnetism_terms[{{2, +1}}] = bfield_spherical[+0] * bfield_spherical[-1];
64  diamagnetism_terms[{{2, -1}}] = bfield_spherical[+0] * bfield_spherical[+1];
65  diamagnetism_terms[{{2, +2}}] = bfield_spherical[-1] * bfield_spherical[-1];
66  diamagnetism_terms[{{2, -2}}] = bfield_spherical[+1] * bfield_spherical[+1];
67 }

122 void SystemOne::initializeBasis() {
123  // If the basis is infinite, throw an error
124  if (range_n.empty() &&
125  (energy_min == std::numeric_limits<double>::lowest() ||
126  energy_max == std::numeric_limits<double>::max())) {
127  throw std::runtime_error(
128  "The number of basis elements is infinite. The basis has to be restricted.");
129  }
130 
131  ////////////////////////////////////////////////////////////////////
132  /// Build one atom states //////////////////////////////////////////
133  ////////////////////////////////////////////////////////////////////
134 
135  // TODO check whether symmetries are applicable
136  // TODO check whether range_j, range_m is half-integer or integer valued
137 
138  float s = 0.5;
139  if (std::isdigit(species.back()) != 0) {
140  s = ((species.back() - '0') - 1) / 2.; // TODO think of a better solution
141  }
142 
143  size_t idx = 0;
144  std::vector<eigen_triplet_t>
145  basisvectors_triplets; // TODO reserve states, basisvectors_triplets,
146  // hamiltonian_triplets
147 
148  std::vector<eigen_triplet_t> hamiltonian_triplets;
149 
150  /// Loop over specified quantum numbers ////////////////////////////
151 
152  std::set<int> range_adapted_n, range_adapted_l;
153  std::set<float> range_adapted_j, range_adapted_m;
154 
155  if (range_n.empty()) {
156  throw std::runtime_error(
157  "The calculation of range_n via energy restrictions is not yet implemented."); // TODO
158  }
159  range_adapted_n = range_n;
160 
161  for (auto n : range_adapted_n) {
162 
163  if (range_l.empty()) {
164  this->range(range_adapted_l, 0, n - 1);
165  } else {
166  range_adapted_l = range_l;
167  }
168  for (auto l : range_adapted_l) {
169  if (l > n - 1 || l < 0) {
170  continue;
171  }
172 
173  if (range_j.empty()) {
174  this->range(range_adapted_j, std::fabs(l - s), l + s);
175  } else {
176  range_adapted_j = range_j;
177  }
178  for (auto j : range_adapted_j) {
179  if (std::fabs(j - l) > s || j < 0) {
180  continue;
181  }
182 
183  double energy = StateOne(species, n, l, j, s).getEnergy(cache);
184  if (!checkIsEnergyValid(energy)) {
185  continue;
186  }
187 
188  if (range_m.empty()) {
189  this->range(range_adapted_m, -j, j);
190  } else {
191  range_adapted_m = range_m;
192  }
193 
194  // Consider rotation symmetry
195  std::set<float> range_allowed_m;
196  if (sym_rotation.count(static_cast<float>(ARB)) == 0) {
197  std::set_intersection(sym_rotation.begin(), sym_rotation.end(),
198  range_adapted_m.begin(), range_adapted_m.end(),
199  std::inserter(range_allowed_m, range_allowed_m.begin()));
200  } else {
201  range_allowed_m = range_adapted_m;
202  }
203 
204  for (auto m : range_allowed_m) {
205  if (std::fabs(m) > j) {
206  continue;
207  }
208 
209  // Create state
210  StateOne state(species, n, l, j, m);
211 
212  // Check whether reflection symmetry can be realized with the states available
213  if (sym_reflection != NA && state.getM() != 0 &&
214  range_allowed_m.count(-state.getM()) == 0) {
215  throw std::runtime_error("The momentum " + std::to_string(-state.getM()) +
216  " required by symmetries cannot be found.");
217  }
218 
219  // Add symmetrized basis vectors
220  this->addSymmetrizedBasisvectors(state, idx, energy, basisvectors_triplets,
221  hamiltonian_triplets, sym_reflection);
222  }
223  }
224  }
225  }
226 
227  /// Loop over user-defined states //////////////////////////////////
228 
229  // Check that the user-defined states are not already contained in the list of states
230  for (const auto &state : states_to_add) {
231  if (states.get<1>().find(state) != states.get<1>().end()) {
232  throw std::runtime_error("The state " + state.str() +
233  " is already contained in the list of states.");
234  }
235  if (!state.isArtificial() && state.getSpecies() != species) {
236  throw std::runtime_error("The state " + state.str() + " is of the wrong species.");
237  }
238  }
239 
240  // Add user-defined states
241  for (const auto &state : states_to_add) {
242  // Get energy of the state
243  double energy = state.isArtificial() ? 0 : state.getEnergy(cache);
244 
245  // In case of artificial states, symmetries won't work
246  auto sym_reflection_local = sym_reflection;
247  auto sym_rotation_local = sym_rotation;
248  if (state.isArtificial()) {
249  if (sym_reflection_local != NA ||
250  sym_rotation_local.count(static_cast<float>(ARB)) == 0) {
251  std::cerr
252  << "WARNING: Only permutation symmetry can be applied to artificial states."
253  << std::endl;
254  }
255  sym_reflection_local = NA;
256  sym_rotation_local = std::set<float>({static_cast<float>(ARB)});
257  }
258 
259  // Consider rotation symmetry
260  if (sym_rotation_local.count(static_cast<float>(ARB)) == 0 &&
261  sym_rotation_local.count(state.getM()) == 0) {
262  continue;
263  }
264 
265  // Check whether reflection symmetry can be realized with the states available
266  if (sym_reflection_local != NA && state.getM() != 0) {
267  auto state_reflected = state.getReflected();
268  if (states_to_add.find(state_reflected) == states_to_add.end()) {
269  throw std::runtime_error("The state " + state_reflected.str() +
270  " required by symmetries cannot be found.");
271  }
272  }
273 
274  // Add symmetrized basis vectors
275  this->addSymmetrizedBasisvectors(state, idx, energy, basisvectors_triplets,
276  hamiltonian_triplets, sym_reflection_local);
277  }
278 
279  /// Build data /////////////////////////////////////////////////////
280 
281  basisvectors.resize(states.size(), idx);
282  basisvectors.setFromTriplets(basisvectors_triplets.begin(), basisvectors_triplets.end());
283  basisvectors_triplets.clear();
284 
285  hamiltonian.resize(idx, idx);
286  hamiltonian.setFromTriplets(hamiltonian_triplets.begin(), hamiltonian_triplets.end());
287  hamiltonian_triplets.clear();
288 }

294 void SystemOne::initializeInteraction() {
295  ////////////////////////////////////////////////////////////////////
296  /// Prepare the calculation of the interaction /////////////////////
297  ////////////////////////////////////////////////////////////////////
298 
299  // Check if something to do
300  double tolerance = 1e-24;
301 
302  std::vector<int> erange, brange;
303  std::vector<std::array<int, 2>> drange;
304 
305  for (const auto &entry : efield_spherical) {
306  if (entry.first < 0) {
307  continue;
308  }
309  if (std::abs(entry.second) > tolerance &&
310  interaction_efield.find(-entry.first) == interaction_efield.end()) {
311  erange.push_back(entry.first);
312  }
313  }
314  for (const auto &entry : bfield_spherical) {
315  if (entry.first < 0) {
316  continue;
317  }
318  if (std::abs(entry.second) > tolerance &&
319  interaction_bfield.find(-entry.first) == interaction_bfield.end()) {
320  brange.push_back(entry.first);
321  }
322  }
323  for (const auto &entry : diamagnetism_terms) {
324  if (entry.first[1] < 0) {
325  continue;
326  }
327  if (diamagnetism && std::abs(entry.second) > tolerance &&
328  interaction_diamagnetism.find(entry.first) == interaction_diamagnetism.end()) {
329  drange.push_back(entry.first);
330  }
331  }
332 
333  // Return if there is nothing to do
334  if (erange.empty() && brange.empty() && drange.empty()) {
335  return;
336  }
337 
338  // Precalculate matrix elements
339  auto states_converted = this->getStates();
340  for (const auto &i : erange) {
341  cache.precalculateElectricMomentum(states_converted, i);
342  if (i != 0) {
343  cache.precalculateElectricMomentum(states_converted, -i);
344  }
345  }
346  for (const auto &i : brange) {
347  cache.precalculateMagneticMomentum(states_converted, i);
348  if (i != 0) {
349  cache.precalculateMagneticMomentum(states_converted, -i);
350  }
351  }
352  for (const auto &i : drange) {
353  cache.precalculateDiamagnetism(states_converted, i[0], i[1]);
354  if (i[1] != 0) {
355  cache.precalculateDiamagnetism(states_converted, i[0], -i[1]);
356  }
357  }
358 
359  ////////////////////////////////////////////////////////////////////
360  /// Calculate the interaction in the canonical basis ///////////////
361  ////////////////////////////////////////////////////////////////////
362 
363  std::unordered_map<int, std::vector<eigen_triplet_t>>
364  interaction_efield_triplets; // TODO reserve
365  std::unordered_map<int, std::vector<eigen_triplet_t>>
366  interaction_bfield_triplets; // TODO reserve
367  std::unordered_map<std::array<int, 2>, std::vector<eigen_triplet_t>>
368  interaction_diamagnetism_triplets; // TODO reserve
369 
370  // Loop over column entries
371  for (const auto &c : states) { // TODO parallelization
372  if (c.state.isArtificial()) {
373  continue;
374  }
375 
376  // Loop over row entries
377  for (const auto &r : states) {
378  if (r.state.isArtificial()) {
379  continue;
380  }
381 
382  // E-field interaction
383  for (const auto &i : erange) {
384  if (i == 0 && r.idx < c.idx) {
385  continue;
386  }
387 
388  if (selectionRulesMultipoleNew(r.state, c.state, 1, i)) {
389  scalar_t value = cache.getElectricDipole(r.state, c.state);
390  this->addTriplet(interaction_efield_triplets[i], r.idx, c.idx, value);
391  break; // because for the other operators, the selection rule for the magnetic
392  // quantum numbers will not be fulfilled
393  }
394  }
395 
396  // B-field interaction
397  for (const auto &i : brange) {
398  if (i == 0 && r.idx < c.idx) {
399  continue;
400  }
401 
402  if (selectionRulesMomentumNew(r.state, c.state, i)) {
403  scalar_t value = cache.getMagneticDipole(r.state, c.state);
404  this->addTriplet(interaction_bfield_triplets[i], r.idx, c.idx, value);
405  break; // because for the other operators, the selection rule for the magnetic
406  // quantum numbers will not be fulfilled
407  }
408  }
409 
410  // Diamagnetic interaction
411  for (const auto &i : drange) {
412  if (i[1] == 0 && r.idx < c.idx) {
413  continue;
414  }
415 
416  if (selectionRulesMultipoleNew(r.state, c.state, i[0], i[1])) {
417  scalar_t value = 1. / (8 * electron_rest_mass) *
418  cache.getDiamagnetism(r.state, c.state, i[0]);
419  this->addTriplet(interaction_diamagnetism_triplets[i], r.idx, c.idx, value);
420  }
421  }
422  }
423  }
424 
425  ////////////////////////////////////////////////////////////////////
426  /// Build and transform the interaction to the used basis //////////
427  ////////////////////////////////////////////////////////////////////
428 
429  for (const auto &i : erange) {
430  interaction_efield[i].resize(states.size(), states.size());
431  interaction_efield[i].setFromTriplets(interaction_efield_triplets[i].begin(),
432  interaction_efield_triplets[i].end());
433  interaction_efield_triplets[i].clear();
434 
435  if (i == 0) {
436  interaction_efield[i] = basisvectors.adjoint() *
437  interaction_efield[i].selfadjointView<Eigen::Lower>() * basisvectors;
438  } else {
439  interaction_efield[i] = basisvectors.adjoint() * interaction_efield[i] * basisvectors;
440  interaction_efield[-i] = std::pow(-1, i) * interaction_efield[i].adjoint();
441  }
442  }
443 
444  for (const auto &i : brange) {
445  interaction_bfield[i].resize(states.size(), states.size());
446  interaction_bfield[i].setFromTriplets(interaction_bfield_triplets[i].begin(),
447  interaction_bfield_triplets[i].end());
448  interaction_bfield_triplets[i].clear();
449 
450  if (i == 0) {
451  interaction_bfield[i] = basisvectors.adjoint() *
452  interaction_bfield[i].selfadjointView<Eigen::Lower>() * basisvectors;
453  } else {
454  interaction_bfield[i] = basisvectors.adjoint() * interaction_bfield[i] * basisvectors;
455  interaction_bfield[-i] = std::pow(-1, i) * interaction_bfield[i].adjoint();
456  }
457  }
458 
459  for (const auto &i : drange) {
460  interaction_diamagnetism[i].resize(states.size(), states.size());
461  interaction_diamagnetism[i].setFromTriplets(interaction_diamagnetism_triplets[i].begin(),
462  interaction_diamagnetism_triplets[i].end());
463  interaction_diamagnetism_triplets[i].clear();
464 
465  if (i[1] == 0) {
466  interaction_diamagnetism[i] = basisvectors.adjoint() *
467  interaction_diamagnetism[i].selfadjointView<Eigen::Lower>() * basisvectors;
468  } else {
469  interaction_diamagnetism[i] =
470  basisvectors.adjoint() * interaction_diamagnetism[i] * basisvectors;
471  interaction_diamagnetism[{{i[0], -i[1]}}] =
472  std::pow(-1, i[1]) * interaction_diamagnetism[i].adjoint();
473  }
474  }
475 }

481 void SystemOne::addInteraction() {
482  // Build the total Hamiltonian
483  double tolerance = 1e-24;
484 
485  if (std::abs(efield_spherical[+0]) > tolerance) {
486  hamiltonian -= interaction_efield[+0] * efield_spherical[+0];
487  }
488  if (std::abs(efield_spherical[-1]) > tolerance) {
489  hamiltonian += interaction_efield[+1] * efield_spherical[-1];
490  }
491  if (std::abs(efield_spherical[+1]) > tolerance) {
492  hamiltonian += interaction_efield[-1] * efield_spherical[+1];
493  }
494  if (std::abs(bfield_spherical[+0]) > tolerance) {
495  hamiltonian -= interaction_bfield[+0] * bfield_spherical[+0];
496  }
497  if (std::abs(bfield_spherical[-1]) > tolerance) {
498  hamiltonian += interaction_bfield[+1] * bfield_spherical[-1];
499  }
500  if (std::abs(bfield_spherical[+1]) > tolerance) {
501  hamiltonian += interaction_bfield[-1] * bfield_spherical[+1];
502  }
503 
504  if (diamagnetism && std::abs(diamagnetism_terms[{{0, +0}}]) > tolerance) {
505  hamiltonian += interaction_diamagnetism[{{0, +0}}] * diamagnetism_terms[{{0, +0}}];
506  }
507  if (diamagnetism && std::abs(diamagnetism_terms[{{2, +0}}]) > tolerance) {
508  hamiltonian -= interaction_diamagnetism[{{2, +0}}] * diamagnetism_terms[{{2, +0}}];
509  }
510  if (diamagnetism && std::abs(diamagnetism_terms[{{2, +1}}]) > tolerance) {
511  hamiltonian +=
512  interaction_diamagnetism[{{2, +1}}] * diamagnetism_terms[{{2, +1}}] * std::sqrt(3);
513  }
514  if (diamagnetism && std::abs(diamagnetism_terms[{{2, -1}}]) > tolerance) {
515  hamiltonian +=
516  interaction_diamagnetism[{{2, -1}}] * diamagnetism_terms[{{2, -1}}] * std::sqrt(3);
517  }
518  if (diamagnetism && std::abs(diamagnetism_terms[{{2, +2}}]) > tolerance) {
519  hamiltonian -=
520  interaction_diamagnetism[{{2, +2}}] * diamagnetism_terms[{{2, +2}}] * std::sqrt(1.5);
521  }
522  if (diamagnetism && std::abs(diamagnetism_terms[{{2, -2}}]) > tolerance) {
523  hamiltonian -=
524  interaction_diamagnetism[{{2, -2}}] * diamagnetism_terms[{{2, -2}}] * std::sqrt(1.5);
525  }
526 }

608  // Combine parameters
609  if (species != dynamic_cast<SystemOne &>(system).species) {
610  throw std::runtime_error(
611  "The value of the variable 'element' must be the same for both systems.");
612  }
613  if (efield != dynamic_cast<SystemOne &>(system).efield) {

618  throw std::runtime_error("The value of the variable 'angle' must be the same for both "

620  // bfield_spherical
621  // is the same,
622  // too
623  }
624  if (diamagnetism != dynamic_cast<SystemOne &>(system).diamagnetism) {
625  throw std::runtime_error(
626  "The value of the variable 'ordermax' must be the same for both systems.");
627  }
628 
629  // Combine symmetries
630  unsigned int num_different_symmetries = 0;
631  if (sym_reflection != dynamic_cast<SystemOne &>(system).sym_reflection) {
632  sym_reflection = NA;
633  ++num_different_symmetries;
634  }
635  if (!std::equal(sym_rotation.begin(), sym_rotation.end(),
636  dynamic_cast<SystemOne &>(system).sym_rotation.begin())) {
637  if (sym_rotation.count(static_cast<float>(ARB)) != 0 ||
638  dynamic_cast<SystemOne &>(system).sym_rotation.count(static_cast<float>(ARB)) != 0) {
639  sym_rotation = {static_cast<float>(ARB)};
640  } else {
641  sym_rotation.insert(dynamic_cast<SystemOne &>(system).sym_rotation.begin(),
642  dynamic_cast<SystemOne &>(system).sym_rotation.end());
643  }
644  ++num_different_symmetries;
645  }
646  if (num_different_symmetries > 1) {
647  std::cerr << "Warning: The systems differ in more than one symmetry. For the combined "
648  "system, the notion of symmetries might be meaningless."
649  << std::endl;
650  }
651 
652  // Clear cached interaction
653  this->deleteInteraction();
654 }

660 void SystemOne::addSymmetrizedBasisvectors(const StateOne &state, size_t &idx, const double &energy,
661  std::vector<eigen_triplet_t> &basisvectors_triplets,
662  std::vector<eigen_triplet_t> &hamiltonian_triplets,
663  parity_t &sym_reflection_local) {
664  // In case of reflection symmetry, skip half of the basis vectors
665  if (sym_reflection_local != NA && state.getM() != 0) {
666  if (state.getM() < 0) {
667  return;
668  }
669  }
670 
671  // Store the energy of the unperturbed one atom state
672  hamiltonian_triplets.emplace_back(idx, idx, energy);
673 
674  // Adapt the normalization if required by symmetries
675  scalar_t value = 1;
676  if (sym_reflection_local != NA && state.getM() != 0) {
677  value /= std::sqrt(2);
678  }
679 
680  // Add an entry to the current basis vector
681  this->addBasisvectors(state, idx, value, basisvectors_triplets);
682 
683  // Add further entries to the current basis vector if required by symmetries
684  if (sym_reflection_local != NA && state.getM() != 0) {
685  value *= std::pow(-1, state.getL() + state.getM() - state.getJ()) *
686  utils::imaginary_unit<scalar_t>();
687  value *= (sym_reflection_local == EVEN) ? 1 : -1;
688  // S_y is invariant under reflection through xz-plane
689  // TODO is the s quantum number of importance here?
690  this->addBasisvectors(state.getReflected(), idx, value, basisvectors_triplets);
691  }
692 
693  ++idx;
694 }
