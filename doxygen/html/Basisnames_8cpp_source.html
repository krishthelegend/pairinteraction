119 void BasisnamesOne::build(StateTwoOld startstate, const std::string &species) {
120  states_initial.push_back(startstate.first()); // TODO correct for idx
121  states_initial.push_back(startstate.second()); // TODO correct for idx
122 
123  conf["species1"] << species;
124  conf["n1"] << startstate.n[0];
125  conf["l1"] << startstate.l[0];
126  conf["j1"] << startstate.j[0];
127  conf["m1"] << startstate.m[0];
128  conf["n2"] << startstate.n[1];
129  conf["l2"] << startstate.l[1];
130  conf["j2"] << startstate.j[1];
131  conf["m2"] << startstate.m[1];
132 
133  std::unordered_set<StateOneOld> names_set;
134 
135  idx_t idx = 0;
136 
137  if (delta_l < 0) {
138  delta_l = std::fmax(startstate.l[0], startstate.l[1]) +
139  std::fmax(startstate.n[0], startstate.n[1]) + delta_n - 1;
140  }
141  if (delta_j < 0) {
142  delta_j = std::fmax(startstate.j[0], startstate.j[1]) +
143  std::fmax(startstate.n[0], startstate.n[1]) + delta_n - 0.5;
144  }
145  if (delta_m < 0) {
146  delta_m = std::fmax(startstate.m[0], startstate.m[1]) +
147  std::fmax(startstate.n[0], startstate.n[1]) + delta_n - 0.5;
148  }
149 
150  // loop over quantum numbers of startstate1
151  for (int n = std::fmax(0, startstate.n[0] - delta_n); n <= startstate.n[0] + delta_n; ++n) {
152  for (int l = std::fmax(0, startstate.l[0] - delta_l);
153  l <= fmin(n - 1, startstate.l[0] + delta_l); ++l) {
154  for (float j = std::fmax(fabs(l - 0.5), startstate.j[0] - delta_j);
155  j <= fmin(l + 0.5, startstate.j[0] + delta_j); ++j) {
156  for (float m = std::fmax(-j, startstate.m[0] - delta_m);
157  m <= fmin(j, startstate.m[0] + delta_m); ++m) {
158  auto result = names_set.insert(StateOneOld(idx, n, l, j, m));
159  if (result.second) {
160  idx++;
161  }
162  }
163  }
164  }
165  }
166 
167  // loop over quantum numbers of startstate2
168  for (int n = std::fmax(0, startstate.n[1] - delta_n); n <= startstate.n[1] + delta_n; ++n) {
169  for (int l = std::fmax(0, startstate.l[1] - delta_l);
170  l <= fmin(n - 1, startstate.l[1] + delta_l); ++l) {
171  for (float j = std::fmax(fabs(l - 0.5), startstate.j[1] - delta_j);
172  j <= fmin(l + 0.5, startstate.j[1] + delta_j); ++j) {
173  for (float m = std::fmax(-j, startstate.m[1] - delta_m);
174  m <= fmin(j, startstate.m[1] + delta_m); ++m) {
175  auto result = names_set.insert(StateOneOld(idx, n, l, j, m));
176  if (result.second) {
177  idx++;
178  }
179  }
180  }
181  }
182  }
183 
184  std::set<StateOneOld> names_ordered(names_set.begin(), names_set.end());

187  dim_ = idx;

189 void BasisnamesOne::build(StateOneOld startstate, const std::string &species) {
190  states_initial.push_back(startstate); // TODO correct for idx
191 
192  conf["species1"] << species;
193  conf["n1"] << startstate.n;
194  conf["l1"] << startstate.l;
195  conf["j1"] << startstate.j;
196  conf["m1"] << startstate.m;
197  conf["n2"] << "";
198  conf["l2"] << "";
199  conf["j2"] << "";
200  conf["m2"] << "";
201 
202  idx_t idx = 0;
203 
204  if (delta_l < 0) {
205  delta_l = startstate.l + startstate.n + delta_n - 1;
206  }
207  if (delta_j < 0) {
208  delta_j = startstate.j + startstate.n + delta_n - 0.5;
209  }
210  if (delta_m < 0) {
211  delta_m = startstate.m + startstate.n + delta_n - 0.5;
212  }
213 
214  // loop over quantum numbers
215  for (int n = std::fmax(0, startstate.n - delta_n); n <= startstate.n + delta_n; ++n) {
216  for (int l = std::fmax(0, startstate.l - delta_l); l <= fmin(n - 1, startstate.l + delta_l);
217  ++l) {
218  for (float j = std::fmax(fabs(l - 0.5), startstate.j - delta_j);
219  j <= fmin(l + 0.5, startstate.j + delta_j); ++j) {
220  for (float m = std::fmax(-j, startstate.m - delta_m);
221  m <= fmin(j, startstate.m + delta_m); ++m) { // TODO
222  names_.emplace_back(idx++, n, l, j, m);
223  }
224  }
225  }
226  }
227 
228  dim_ = idx;
