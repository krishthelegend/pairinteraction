33 inline boost::variant<char, int> getMomentumLabel(int l) {
34  switch (l) {
35  case 0:
36  return 'S';
37  case 1:
38  return 'P';
39  case 2:
40  return 'D';
41  case 3:
42  return 'F';
43  case 4:
44  return 'G';
45  case 5:
46  return 'H';
47  case 6:
48  return 'I';
49  default:
50  return l;
51  }
52 }

60 StateOne::StateOne(std::string species, int n, int l, float j, float m)
61  : species(std::move(species)), n(n), l(l), j(j), m(m) {
62  this->analyzeSpecies();
63  hashvalue = 0;
64  boost::hash_combine(hashvalue, this->getSpecies());
65  boost::hash_combine(hashvalue, this->getN());
66  boost::hash_combine(hashvalue, this->getL());
67  boost::hash_combine(hashvalue, this->getJ());
68  boost::hash_combine(hashvalue, this->getM());
69 }

77 std::ostream &operator<<(std::ostream &out, const StateOne &state) {
78  out << "|";
79  if (state.isArtificial()) {
80  out << state.getLabel();
81  } else {
82  out << state.getSpecies() << ", " << state.getN() << " " << getMomentumLabel(state.getL())
83  << "_";
84  if (std::ceil(state.getJ()) == state.getJ()) {
85  out << state.getJ() << ", ";
86  out << "mj=" << state.getM();
87  } else {
88  out << 2 * state.getJ() << "/2, ";
89  out << "mj=" << 2 * state.getM() << "/2";
90  }
91  }
92  out << ">";
93  return out;
94 }

183 // Utility methods
184 void StateOne::analyzeSpecies() {
185  s = 0.5;
186  element = species;
187  if (std::isdigit(species.back()) != 0) {
188  s = ((species.back() - '0') - 1) / 2.;
189  element = species.substr(0, species.size() - 1);
190  }
191 }
