33 double PerturbativeInteraction::getC6(const StateTwo &state, double deltaN) {
34  double C6 = 0;
35 
36  for (int n0 = state.getN(0) - deltaN; n0 <= state.getN(0) + deltaN; ++n0) {
37 
38  for (int n1 = state.getN(1) - deltaN; n1 <= state.getN(1) + deltaN; ++n1) {
39 
40  std::vector<int> array_l0;
41  if (state.getL(0) > 0) {
42  array_l0.push_back(state.getL(0) - 1);
43  }
44  if (state.getL(0) < n0 - 1) {
45  array_l0.push_back(state.getL(0) + 1);
46  }
47  for (int l0 : array_l0) {
48 
49  std::vector<int> array_l1;
50  if (state.getL(1) > 0) {
51  array_l1.push_back(state.getL(1) - 1);
52  }
53  if (state.getL(1) < n1 - 1) {
54  array_l1.push_back(state.getL(1) + 1);
55  }
56  for (int l1 : array_l1) {
57 
58  std::set<float> array_j0;
59  if (std::abs(std::abs(l0 - state.getS(0)) - state.getJ(0)) < 2) {
60  array_j0.insert(std::abs(l0 - state.getS(0)));
61  }
62  if (std::abs(l0 + state.getS(0) - state.getJ(0)) < 2) {
63  array_j0.insert(l0 + state.getS(0));
64  }
65  for (float j0 : array_j0) {
66 
67  std::set<float> array_j1;
68  if (std::abs(std::abs(l1 - state.getS(1)) - state.getJ(1)) < 2) {
69  array_j1.insert(std::abs(l1 - state.getS(1)));
70  }
71  if (std::abs(l1 + state.getS(1) - state.getJ(1)) < 2) {
72  array_j1.insert(l1 + state.getS(1));
73  }
74  for (float j1 : array_j1) {
75 
76  for (auto &q : array_q) {
77  // q = final.m - initial.m
78 
79  float m0 = state.getM(0) + q[0];
80  if (std::abs(m0) > j0) {
81  continue;
82  }
83 
84  float m1 = state.getM(1) + q[1];
85  if (std::abs(m1) > j1) {
86  continue;
87  }
88 
89  StateTwo state_virtual =
90  StateTwo(state.getSpecies(), {{n0, n1}}, {{l0, l1}}, {{j0, j1}},
91  {{m0, m1}});
92 
93  double energydiff =
94  state.getEnergy(cache) - state_virtual.getEnergy(cache);
95 
96  C6 +=
97  std::pow(
98  coulombs_constant *
99  array_angle_term[3 * (q[0] + 1) + (q[1] + 1)] *
100  cache.getElectricDipole(state_virtual.getFirstState(),
101  state.getFirstState()) *
102  cache.getElectricDipole(state_virtual.getSecondState(),
103  state.getSecondState()),
104  2) /
105  energydiff; // getMultipole(final, inital, 1)
106  }
107  }
108  }
109  }
110  }
111  }
112  }
113 
114  return C6;
115 }

117 eigen_dense_double_t PerturbativeInteraction::getC6(const std::vector<StateTwo> &states,
118  double deltaN) {
119  eigen_dense_double_t C6_matrix = eigen_dense_double_t::Zero(states.size(), states.size());
120 
121  std::unordered_set<StateTwo> set_states(states.begin(), states.end());
122 
123  for (size_t idx_row = 0; idx_row < states.size(); ++idx_row) {
124  auto &state_row = states[idx_row];
125 
126  for (size_t idx_col = idx_row; idx_col < states.size(); ++idx_col) {
127  auto &state_col = states[idx_col];
128 
129  if (state_row.getS(0) != state_col.getS(0) || state_row.getS(1) != state_col.getS(1)) {
130  continue;
131  }
132  auto s = state_col.getS();
133 
134  if (state_row.getSpecies(0) != state_col.getSpecies(0) ||
135  state_row.getSpecies(1) != state_col.getSpecies(1)) {
136  continue;
137  }
138  auto species = state_col.getSpecies();
139 
140  int n0_min = std::min(state_row.getN(0), state_col.getN(0));
141  int n0_max = std::max(state_row.getN(0), state_col.getN(0));
142  int n1_min = std::min(state_row.getN(1), state_col.getN(1));
143  int n1_max = std::max(state_row.getN(1), state_col.getN(1));
144 
145  double C6 = 0;
146 
147  for (int n0 = n0_min - deltaN; n0 <= n0_max + deltaN; ++n0) {
148 
149  for (int n1 = n1_min - deltaN; n1 <= n1_max + deltaN; ++n1) {
150 
151  std::vector<int> array_l0;
152  if (state_row.getL(0) == state_col.getL(0)) {
153  if (state_col.getL(0) < n0 - 1) {
154  array_l0.push_back(state_col.getL(0) + 1);
155  }
156  if (state_col.getL(0) > 0) {
157  array_l0.push_back(state_col.getL(0) - 1);
158  }
159  } else if (state_row.getL(0) + 2 == state_col.getL(0) &&
160  state_col.getL(0) < n0 - 1) {
161  array_l0.push_back(state_col.getL(0) + 1);
162  } else if (state_row.getL(0) - 2 == state_col.getL(0) &&
163  state_col.getL(0) > 0) {
164  array_l0.push_back(state_col.getL(0) - 1);
165  }
166  for (int l0 : array_l0) {
167 
168  std::vector<int> array_l1;
169  if (state_row.getL(1) == state_col.getL(1)) {
170  if (state_col.getL(1) < n1 - 1) {
171  array_l1.push_back(state_col.getL(1) + 1);
172  }
173  if (state_col.getL(1) > 0) {
174  array_l1.push_back(state_col.getL(1) - 1);
175  }
176  } else if (state_row.getL(1) + 2 == state_col.getL(1) &&
177  state_col.getL(1) < n1 - 1) {
178  array_l1.push_back(state_col.getL(1) + 1);
179  } else if (state_row.getL(1) - 2 == state_col.getL(1) &&
180  state_col.getL(1) > 0) {
181  array_l1.push_back(state_col.getL(1) - 1);
182  }
183  for (int l1 : array_l1) {
184 
185  std::set<float> array_j0;
186  if (std::abs(std::abs(l0 - s[0]) - state_col.getJ(0)) < 2 &&
187  std::abs(std::abs(l0 - s[0]) - state_row.getJ(0)) < 2) {
188  array_j0.insert(std::abs(l0 - s[0]));
189  }
190  if (std::abs(l0 + s[0] - state_col.getJ(0)) < 2 &&
191  std::abs(l0 + s[0] - state_row.getJ(0)) < 2) {
192  array_j0.insert(l0 + s[0]);
193  }
194  for (float j0 : array_j0) {
195 
196  std::set<float> array_j1;
197  if (std::abs(std::abs(l1 - s[1]) - state_col.getJ(1)) < 2 &&
198  std::abs(std::abs(l1 - s[1]) - state_row.getJ(1)) < 2) {
199  array_j1.insert(std::abs(l1 - s[1]));
200  }
201  if (std::abs(l1 + s[1] - state_col.getJ(1)) < 2 &&
202  std::abs(l1 + s[1] - state_row.getJ(1)) < 2) {
203  array_j1.insert(l1 + s[1]);
204  }
205  for (float j1 : array_j1) {
206 
207  for (auto &idx : array_q) {
208  int q0_forth = idx[0]; // q = final.m - initial.m
209  int q1_forth = idx[1];
210 
211  float m0 = state_col.getM(0) + q0_forth;
212  if (std::abs(m0) > j0) {
213  continue;
214  }
215 
216  float m1 = state_col.getM(1) + q1_forth;
217  if (std::abs(m1) > j1) {
218  continue;
219  }
220 
221  int q0_back = state_row.getM(0) - m0;
222  int q1_back = state_row.getM(1) - m1;
223  if (std::abs(q0_back) > 1 || std::abs(q1_back) > 1) {
224  continue;
225  }
226  if (array_q.size() == 3 && q0_back + q1_back != 0) {
227  continue;
228  }
229 
230  StateTwo state_virtual =
231  StateTwo(species, {{n0, n1}}, {{l0, l1}}, {{j0, j1}},
232  {{m0, m1}});
233  if (set_states.find(state_virtual) != set_states.end()) {
234  continue;
235  }
236 
237  double energydiff_row = state_row.getEnergy(cache) -
238  state_virtual.getEnergy(cache);
239 
240  double energydiff_col = state_col.getEnergy(cache) -
241  state_virtual.getEnergy(cache);
242 
243  C6 += coulombs_constant *
244  array_angle_term[3 * (q0_back + 1) + (q1_back + 1)] *
245  cache.getElectricDipole(state_row.getFirstState(),
246  state_virtual.getFirstState()) *
247  cache.getElectricDipole(
248  state_row.getSecondState(),
249  state_virtual.getSecondState()) *
250  coulombs_constant *
251  array_angle_term[3 * (q0_forth + 1) + (q1_forth + 1)] *
252  cache.getElectricDipole(state_virtual.getFirstState(),
253  state_col.getFirstState()) *
254  cache.getElectricDipole(state_virtual.getSecondState(),
255  state_col.getSecondState()) *
256  0.5 *
257  (1 / energydiff_row +
258  1 / energydiff_col); // getMultipole(final, inital, 1)
259  }
260  }
261  }
262  }
263  }
264  }
265  }
266 
267  C6_matrix(idx_row, idx_col) = C6;
268  }
269  }
270 
271  return C6_matrix.selfadjointView<Eigen::Upper>();
272 }

274 eigen_dense_double_t PerturbativeInteraction::getC3(const std::vector<StateTwo> &states) {
275  eigen_dense_double_t C3_matrix = eigen_dense_double_t::Zero(states.size(), states.size());
276 
277  for (size_t idx_row = 0; idx_row < states.size(); ++idx_row) {
278  auto &state_row = states[idx_row];
279 
280  for (size_t idx_col = idx_row + 1; idx_col < states.size(); ++idx_col) {
281  auto &state_col = states[idx_col];
282 
283  int q0 = state_row.getM(0) - state_col.getM(0);
284  int q1 = state_row.getM(1) - state_col.getM(1);
285 
286  if (array_q.size() == 3 && q0 + q1 != 0) {
287  continue;
288  }
289  if (std::abs(q0) > 1 || std::abs(q1) > 1) {
290  continue;
291  }
292 
293  C3_matrix(idx_row, idx_col) = coulombs_constant *
294  array_angle_term[3 * (q0 + 1) + (q1 + 1)] *
295  cache.getElectricDipole(state_row.getFirstState(), state_col.getFirstState()) *
296  cache.getElectricDipole(state_row.getSecondState(), state_col.getSecondState());
297  }
298  }
299 
300  return C3_matrix.selfadjointView<Eigen::Upper>();
301 }

316 void PerturbativeInteraction::initializeAngleTerms(double angle) {
317  if (angle == 0) {
318  array_q.reserve(3);
319 
320  array_q.push_back({{0, 0}});
321  array_angle_term[3 * (0 + 1) + (0 + 1)] = -2;
322 
323  array_q.push_back({{1, -1}});
324  array_angle_term[3 * (1 + 1) + (-1 + 1)] = -1;
325 
326  array_q.push_back({{-1, 1}});
327  array_angle_term[3 * (-1 + 1) + (1 + 1)] = -1;
328  } else {
329  array_q.reserve(9);
330 
331  array_q.push_back({{0, 0}});
332  array_angle_term[3 * (0 + 1) + (0 + 1)] = 1. - 3. * std::pow(std::cos(angle), 2);
333 
334  array_q.push_back({{1, -1}});
335  array_angle_term[3 * (1 + 1) + (-1 + 1)] = -1 + 1.5 * std::pow(std::sin(angle), 2);
336 
337  array_q.push_back({{-1, 1}});
338  array_angle_term[3 * (-1 + 1) + (1 + 1)] = -1 + 1.5 * std::pow(std::sin(angle), 2);
339 
340  array_q.push_back({{1, 1}});
341  array_angle_term[3 * (1 + 1) + (1 + 1)] = -1.5 * std::pow(std::sin(angle), 2);
342 
343  array_q.push_back({{-1, -1}});
344  array_angle_term[3 * (-1 + 1) + (-1 + 1)] = -1.5 * std::pow(std::sin(angle), 2);
345 
346  array_q.push_back({{0, 1}});
347  array_angle_term[3 * (0 + 1) + (1 + 1)] =
348  3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
349 
350  array_q.push_back({{1, 0}});
351  array_angle_term[3 * (1 + 1) + (0 + 1)] =
352  3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
353 
354  array_q.push_back({{0, -1}});
355  array_angle_term[3 * (0 + 1) + (-1 + 1)] =
356  -3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
357 
358  array_q.push_back({{-1, 0}});
359  array_angle_term[3 * (-1 + 1) + (0 + 1)] =
360  -3. / std::sqrt(2) * std::sin(angle) * std::cos(angle);
361  }
