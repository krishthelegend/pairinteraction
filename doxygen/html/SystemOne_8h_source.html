110  template <class T>
111  void addRotated(const StateOne &state, const size_t &idx,
112  std::vector<Eigen::Triplet<T>> &triplets, WignerD &wigner, const double &alpha,
113  const double &beta, const double &gamma) {
114  // Check whether the angles are compatible to the used data type
115  double tolerance = 1e-16;
116  if (std::is_same<T, double>::value &&
117  std::abs(std::remainder(alpha, 2 * M_PI)) > tolerance) {
118  throw std::runtime_error(
119  "If the Euler angle alpha is not a multiple of 2 pi, the Wigner D matrix element "
120  "is complex and cannot be converted to double.");
121  }
122  if (std::is_same<T, double>::value &&
123  std::abs(std::remainder(gamma, 2 * M_PI)) > tolerance) {
124  throw std::runtime_error(
125  "If the Euler angle gamma is not a multiple of 2 pi, the Wigner D matrix element "
126  "is complex and cannot be converted to double.");
127  }
128 
129  // Add rotated triplet entries
130  for (float m = -state.getJ(); m <= state.getJ(); ++m) {
131  StateOne newstate(state.getSpecies(), state.getN(), state.getL(), state.getJ(), m);
132  auto state_iter = states.get<1>().find(newstate);
133 
134  if (state_iter != states.get<1>().end()) {
135  // We calculate the matrix element <m|d_mM|state.getM()>|m>. Note that we must
136  // invert the angles since they are given for a passive rotation, i.e. rotation of
137  // the coordinate system, and the WignerD matrix is for an active rotation of spins.
138  auto val =
139  utils::convert<T>(wigner(state.getJ(), m, state.getM(), -gamma, -beta, -alpha));
140  triplets.push_back(Eigen::Triplet<T>(state_iter->idx, idx, val));
141  } else {
142  std::cerr << "Warning: Incomplete rotation because the basis is lacking some "
143  "Zeeman levels."
144  << std::endl;
145  }
146  }
147  }
