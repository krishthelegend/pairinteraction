210  eigen_vector_double_t getOverlap(const std::vector<T> &generalizedstates, double alpha,
211  double beta, double gamma) {
212  // Build basis
213  this->buildBasis();
214 
215  // Determine indices of the specified states
216  std::vector<size_t> states_indices;
217  states_indices.reserve(generalizedstates.size());
218 
219  for (const auto &searched_state : generalizedstates) {
220  if (utils::is_true(searched_state.isGeneralized())) {
221  for (const auto &state : states) {
222  if (state.state ^ searched_state) { // Check whether state.state is contained in
223  // searched_state
224  states_indices.push_back(state.idx);
225  }
226  }
227  } else {
228  auto state_iter = states.template get<1>().find(searched_state);
229  if (state_iter != states.template get<1>().end()) {
230  states_indices.push_back(state_iter->idx);
231  }
232  }
233  }
234 
235  // Get the overlap
236  return this->getOverlap(states_indices, alpha, beta, gamma);
237  }

239  eigen_vector_double_t getOverlap(const std::vector<size_t> &states_indices, double alpha,
240  double beta, double gamma) {
241  // Build basis
242  this->buildBasis();
243 
244  // Build state vectors
245  eigen_sparse_t overlap_states;
246  if (alpha == 0 && beta == 0 && gamma == 0) {
247  std::vector<eigen_triplet_t> overlap_states_triplets;
248  overlap_states_triplets.reserve(states_indices.size());
249 
250  size_t current = 0;
251  for (auto const &idx : states_indices) {
252  overlap_states_triplets.emplace_back(idx, current++, 1);
253  }
254 
255  overlap_states.resize(states.size(), states_indices.size());
256  overlap_states.setFromTriplets(overlap_states_triplets.begin(),
257  overlap_states_triplets.end());
258  } else {
259  overlap_states = this->rotateStates(states_indices, alpha, beta, gamma);
260  }
261 
262  // Calculate overlap
263  eigen_sparse_t product = basisvectors.adjoint() * overlap_states;
264  eigen_vector_double_t overlap = eigen_vector_double_t::Zero(product.rows());
265  for (int k = 0; k < product.outerSize(); ++k) {
266  for (eigen_iterator_t triple(product, k); triple; ++triple) {
267  overlap[triple.row()] += std::pow(std::abs(triple.value()), 2);
268  }
269  }
270 
271  return overlap;
272  }

357  std::array<std::vector<size_t>, 2> getConnections(SystemBase<T> &system_to, double threshold) {
358  // Build basis
359  this->buildBasis();
360  system_to.buildBasis();
361 
362  double threshold_sqrt = std::sqrt(threshold);
363 
364  // Calculate transformator between the set of states
365  std::vector<eigen_triplet_t> triplets_transformator;
366  triplets_transformator.reserve(std::min(states.size(), system_to.states.size()));
367 
368  for (const auto &s : system_to.states) {
369  auto state_iter = states.template get<1>().find(s.state);
370  if (state_iter != states.template get<1>().end()) {
371  size_t idx_from = state_iter->idx;
372  triplets_transformator.emplace_back(idx_from, s.idx, 1);
373  }
374  }
375 
376  eigen_sparse_t transformator(states.size(), system_to.states.size());
377  transformator.setFromTriplets(triplets_transformator.begin(), triplets_transformator.end());
378 
379  // Calculate overlap
380  eigen_sparse_double_t overlap_sqrt =
381  (basisvectors.adjoint() * transformator * system_to.basisvectors).cwiseAbs();
382 
383  // Determine the indices of the maximal values within the rows
384  std::vector<size_t> rows_with_maxval;
385  rows_with_maxval.reserve(overlap_sqrt.cols());
386 
387  for (int k = 0; k < overlap_sqrt.outerSize(); ++k) {
388  double maxval = -1;
389  size_t row_with_maxval;
390 
391  for (eigen_iterator_double_t triple(overlap_sqrt, k); triple; ++triple) {
392  if (triple.value() > maxval) {
393  row_with_maxval = triple.row();
394  maxval = triple.value();
395  }
396  }
397 
398  rows_with_maxval.push_back(row_with_maxval);
399  }
400 
401  // Determine the maximal values within the columns and construct connections
402  eigen_sparse_double_t overlap_sqrt_transposed = overlap_sqrt.transpose();
403  std::array<std::vector<size_t>, 2> connections;
404  connections[0].reserve(std::max(overlap_sqrt.rows(), overlap_sqrt.cols()));
405  connections[1].reserve(std::max(overlap_sqrt.rows(), overlap_sqrt.cols()));
406 
407  for (int k = 0; k < overlap_sqrt_transposed.outerSize(); ++k) { // cols
408  double maxval = threshold_sqrt;
409  size_t idx_from;
410  size_t idx_to;
411 
412  for (eigen_iterator_double_t triple(overlap_sqrt_transposed, k); triple; ++triple) {
413  if (triple.value() > maxval) {
414  idx_from = triple.col();
415  idx_to = triple.row();
416  maxval = triple.value();
417  }
418  }
419 
420  if (maxval > threshold_sqrt && rows_with_maxval[idx_to] == idx_from) {
421  connections[0].push_back(idx_from);
422  connections[1].push_back(idx_to);
423  }
424  }
425 
426  connections[0].shrink_to_fit();
427  connections[1].shrink_to_fit();
428 
429  return connections;
430  }

486  void buildBasis() {
487  // Check variables for consistency
488  if (((hamiltonian.size() == 0) != states.empty()) ||
489  ((hamiltonian.size() == 0) != (basisvectors.size() == 0))) {
490  throw std::runtime_error("Inconsistent variables at " + std::string(__FILE__) + ":" +
491  std::to_string(__LINE__) + ".");
492  }
493 
494  // In case of no new basis restrictions and already initialized basis, there is nothing to
495  // do
496  if (!states.empty() && states_to_add.empty() && range_n.empty() && range_l.empty() &&
497  range_j.empty() && range_m.empty() &&
498  energy_min == std::numeric_limits<double>::lowest() &&

529  void diagonalize(double energy_lower_bound, double energy_upper_bound, double threshold) {
530 #ifdef WITH_INTEL_MKL
531  this->buildHamiltonian();
532 
533  // Check if already diagonal
534  if (checkIsDiagonal(hamiltonian)) {
535  return;
536  }
537 
538  // Prepare the mkl library
539  mkl_set_threading_layer(MKL_THREADING_SEQUENTIAL);
540  mkl_set_interface_layer(MKL_INTERFACE_LP64);
541 
542  // Estimate number of found eigenvalues
543  std::vector<double> diagonal_max(hamiltonian.outerSize(), 0);
544  std::vector<double> diagonal_val;
545  std::vector<int> diagonal_idx;
546  diagonal_val.reserve(hamiltonian.outerSize());
547  diagonal_idx.reserve(hamiltonian.outerSize());
548  for (int k = 0; k < hamiltonian.outerSize(); ++k) {
549  double val = 0;
550  for (eigen_iterator_t triple(hamiltonian, k); triple; ++triple) {
551  if (triple.row() == triple.col()) {
552  val = std::real(triple.value());
553  } else if (triple.row() != triple.col()) {
554  diagonal_max[k] = std::max(diagonal_max[k], std::abs(triple.value()));
555  }
556  }
557  diagonal_idx.push_back(k);
558  diagonal_val.push_back(val);
559  }
560 
561  MKL_INT m0 =
562  std::count_if(diagonal_val.begin(), diagonal_val.end(),
563  [&energy_lower_bound, &energy_upper_bound](const double &val) {
564  return (val < energy_upper_bound) && (val > energy_lower_bound);
565  }) +
566  std::count_if(diagonal_idx.begin(), diagonal_idx.end(),
567  [&energy_lower_bound, &energy_upper_bound, &diagonal_val,
568  &diagonal_max](const int &i) {
569  return ((diagonal_val[i] >= energy_upper_bound) ||
570  (diagonal_val[i] <= energy_lower_bound)) &&
571  (diagonal_val[i] < energy_upper_bound + diagonal_max[i]) &&
572  (diagonal_val[i] > energy_lower_bound - diagonal_max[i]);
573  });
574 
575  // Inplace conversion of the Hamiltonian to CSR with one-based indexing
576  hamiltonian = hamiltonian.transpose();
577  hamiltonian.makeCompressed();
578  for (int i = 0; i < hamiltonian.rows() + 1; ++i) {
579  hamiltonian.outerIndexPtr()[i] += 1;
580  }
581  for (int i = 0; i < hamiltonian.nonZeros(); ++i) {
582  hamiltonian.innerIndexPtr()[i] += 1;
583  }
584 
585  // Set default parameters for the diagonalization as described at
586  // https://software.intel.com/en-us/mkl-developer-reference-c-extended-eigensolver-input-parameters.
587 
588  std::vector<MKL_INT> fpm(128);
589  feastinit(&fpm[0]);
590  fpm[0] = 1; // enables terminal output
591  fpm[1] = 6; // number of contour points
592  fpm[26] = 0; // disables matrix checker
593  fpm[3] = 5; // maximum number of refinement loops allowed
594  if (threshold != 0) {
595  // Adapt the error trace stopping criteria (10-fpm[2])
596  fpm[2] = std::min(std::round(-std::log10(threshold)), 12.);
597  }
598 
599  // Do the diagonalization
600  {
601  MKL_INT n = hamiltonian.rows(); // size of the matrix
602  MKL_INT m; // will contain the number of eigenvalues
603  std::vector<scalar_t> x(m0 * n); // the first m columns will contain the eigenvectors
604  {
605  std::vector<double> e(m0); // will contain the first m eigenvalues
606  {
607  char uplo = 'F'; // full matrix is stored
608  MKL_INT info; // will contain return codes
609  double epsout; // will contain relative error
610  MKL_INT loop; // will contain number of used refinement
611  std::vector<double> res(m0); // will contain the residual errors
612 
613  this->feast_csrev(&uplo, &n, hamiltonian.valuePtr(),
614  hamiltonian.outerIndexPtr(), hamiltonian.innerIndexPtr(),
615  &fpm[0], &epsout, &loop, &energy_lower_bound,
616  &energy_upper_bound, &m0, &e[0], &x[0], &m, &res[0], &info);
617  }
618 
619  // Build the new hamiltonian
620  hamiltonian.resize(m, m);
621  hamiltonian.setZero();
622  hamiltonian.reserve(m);
623  for (int idx = 0; idx < m; ++idx) {
624  hamiltonian.insert(idx, idx) = e[idx];
625  }
626  hamiltonian.makeCompressed();
627  }
628 
629  // Transform the basis vectors
630  eigen_sparse_t evecs = Eigen::Map<eigen_dense_t>(&x[0], n, m).sparseView();
631  if (threshold == 0) {
632  basisvectors = basisvectors * evecs;
633  } else {
634  basisvectors = (basisvectors * evecs).pruned(threshold, 1);
635  }
636  }
637 #else // WITH_INTEL_MKL
638  (void)energy_lower_bound;
639  (void)energy_upper_bound;
640  (void)threshold;
641  throw std::runtime_error(

761  void add(SystemBase<T> &system) {
762  // --- Build bases ---
763  this->buildBasis();
764  system.buildBasis();
765 
766  size_t size_this = basisvectors.cols();
767  size_t size_other = system.basisvectors.cols();
768 
769  // --- Combine system specific variables ---
770  this->incorporate(system);
771 
772  // --- Combine universal variables ---
773  if (memory_saving != system.memory_saving) {
774  throw std::runtime_error(
775  "The value of the variable 'memory_saving' must be the same for both systems.");
776  }
777  if (is_interaction_already_contained != system.is_interaction_already_contained) {
778  throw std::runtime_error("The value of the variable 'is_interaction_already_contained' "
779  "must be the same for both systems.");
780  }
781  if (is_new_hamiltonian_required != system.is_new_hamiltonian_required) {
782  throw std::runtime_error(
783  "The value of the variable 'is_new_hamiltonian_required' must be the same "
784  "for both systems.");
785  }
786 
787  // --- Combine states and build transformators ---
788  eigen_sparse_t transformator;
789  {
790  std::vector<eigen_triplet_t> transformator_triplets;
791  transformator_triplets.reserve(system.states.size());
792 
793  for (const auto &entry : system.states) {
794  auto state_iter = states.template get<1>().find(entry.state);
795 
796  size_t newidx = states.size();
797  if (state_iter == states.template get<1>().end()) {
798  states.push_back(enumerated_state<T>(newidx, entry.state));
799  } else {
800  newidx = state_iter->idx;
801  }
802 
803  transformator_triplets.emplace_back(newidx, entry.idx, 1);
804  }
805 
806  transformator.resize(states.size(), system.basisvectors.rows());
807  transformator.setFromTriplets(transformator_triplets.begin(),
808  transformator_triplets.end());
809  }
810 
811  // --- Combine basisvectors and basisvectors_unperturbed_cache ---
812  basisvectors.conservativeResize(states.size(), size_this + size_other);
813  basisvectors.rightCols(size_other) = transformator * system.basisvectors;
814 
815  if ((basisvectors_unperturbed_cache.size() != 0) !=
816  (system.basisvectors_unperturbed_cache.size() != 0)) {
817  throw std::runtime_error("Inconsistent variables at " + std::string(__FILE__) + ":" +
818  std::to_string(__LINE__) + ".");
819  }
820 
821  if (basisvectors_unperturbed_cache.size() != 0) {
822  basisvectors_unperturbed_cache.conservativeResize(
823  states.size(),
824  basisvectors_unperturbed_cache.cols() +
825  system.basisvectors_unperturbed_cache.cols());
826  basisvectors_unperturbed_cache.rightCols(system.basisvectors_unperturbed_cache.cols()) =
827  transformator * system.basisvectors_unperturbed_cache;
828  }
829 
830  // --- Check if basis vectors are orthogonal ---
831  eigen_sparse_t tmp =
832  (basisvectors.leftCols(size_this).adjoint() * basisvectors.rightCols(size_other))
833  .pruned(1e-12, 1);
834  if (tmp.nonZeros() != 0) {
835  throw std::runtime_error(
836  "Two systems cannot be combined if their basis vectors are not orthogonal.");
837  }
838 
839  // --- Combine hamiltonian and hamiltonian_unperturbed_cache ---
840  eigen_sparse_t shifter(hamiltonian.rows() + system.hamiltonian.rows(),
841  system.hamiltonian.rows());
842  {
843  std::vector<eigen_triplet_t> shifter_triplets;
844  shifter_triplets.reserve(system.hamiltonian.rows());
845 
846  for (size_t idx = 0; idx < system.hamiltonian.rows(); ++idx) {
847  shifter_triplets.emplace_back(hamiltonian.rows() + idx, idx, 1);
848  }
849 
850  shifter.setFromTriplets(shifter_triplets.begin(), shifter_triplets.end());
851  }
852 
853  hamiltonian.conservativeResize(hamiltonian.rows() + system.hamiltonian.rows(),
854  hamiltonian.cols() + system.hamiltonian.cols());
855  hamiltonian.rightCols(system.hamiltonian.cols()) = shifter * system.hamiltonian;
856 
857  if ((hamiltonian_unperturbed_cache.size() != 0) !=
858  (system.hamiltonian_unperturbed_cache.size() != 0)) {
859  throw std::runtime_error("Inconsistent variables at " + std::string(__FILE__) + ":" +
860  std::to_string(__LINE__) + ".");
861  }
862 
863  if (hamiltonian_unperturbed_cache.size() != 0) {
864  hamiltonian_unperturbed_cache.conservativeResize(
865  hamiltonian_unperturbed_cache.rows() + system.hamiltonian_unperturbed_cache.rows(),
866  hamiltonian_unperturbed_cache.cols() + system.hamiltonian_unperturbed_cache.cols());
867  hamiltonian_unperturbed_cache.rightCols(system.hamiltonian_unperturbed_cache.cols()) =
868  shifter * system.hamiltonian_unperturbed_cache;
869  }

900  void applySchriefferWolffTransformation(SystemBase<T> &system0) {
901  this->diagonalize();
902  system0.buildHamiltonian();
903 
904  // Check that system, on which applySchriefferWolffTransformation() is called, is unitary
905  if (!this->checkIsUnitary(basisvectors)) {
906  throw std::runtime_error("The system, on which applySchriefferWolffTransformation() is "
907  "called, is not unitary. Call unitarize() on the system.");
908  }
909 
910  // Check that system0, i.e. the unperturbed system, is unitary
911  if (!this->checkIsUnitary(system0.basisvectors)) {
912  throw std::runtime_error(
913  "The unperturbed system passed to applySchriefferWolffTransformation() is not "
914  "unitary. Call unitarize() on the system.");
915  }
916 
917  // Check that the unperturbed system is diagonal
918  if (!this->checkIsDiagonal(system0.hamiltonian)) {
919  throw std::runtime_error("The unperturbed system passed to "
920  "applySchriefferWolffTransformation() is not diagonal.");
921  }
922 
923  // --- Express the basis vectors of system0 as a linearcombination of all states ---
924 
925  // Combine states and build transformators
926  eigen_sparse_t transformator;
927  {
928  std::vector<eigen_triplet_t> transformator_triplets;
929  transformator_triplets.reserve(system0.states.size());
930 
931  for (const auto &entry : system0.states) {
932  auto state_iter = states.template get<1>().find(entry.state);
933 
934  // Check that all the states of system0 occur (since we already checked that the
935  // number of states is the same, this ensures that all the states are the same)
936  if (state_iter == states.template get<1>().end()) {
937  throw std::runtime_error(
938  "The unperturbed system contains states that are not occuring.");
939  }
940  size_t newidx = state_iter->idx;
941 
942  transformator_triplets.emplace_back(newidx, entry.idx, 1);
943  }
944 
945  transformator.resize(states.size(), system0.states.size());
946  transformator.setFromTriplets(transformator_triplets.begin(),
947  transformator_triplets.end());
948  }
949 
950  // Get basisvectors of system0
951  eigen_sparse_t low_energy_basis0 = transformator * system0.basisvectors;
952 
953  // --- Find basisvectors which corresponds to the basis vectors of system0 ---
954 
955  // Get overlaps between basis vectors
956  eigen_vector_double_t overlap = (basisvectors.adjoint() * low_energy_basis0).cwiseAbs2() *
957  eigen_vector_double_t::Ones(low_energy_basis0.cols());
958 
959  // Get indices of the low_energy_basis0.cols() largest entries and build transformator
960  {
961  std::vector<int> indices(basisvectors.cols());
962  std::iota(indices.begin(), indices.end(), 0);
963  std::nth_element(indices.begin(), indices.begin() + low_energy_basis0.cols(),
964  indices.end(),
965  [&overlap](int a, int b) { return overlap[a] > overlap[b]; });
966 
967  std::vector<eigen_triplet_t> transformator_triplets;
968  transformator_triplets.reserve(low_energy_basis0.cols());
969 
970  for (int i = 0; i < low_energy_basis0.cols(); ++i) {
971  transformator_triplets.emplace_back(indices[i], i, 1);
972  }
973 
974  transformator.resize(basisvectors.cols(), low_energy_basis0.cols());
975  transformator.setFromTriplets(transformator_triplets.begin(),
976  transformator_triplets.end());
977  }
978 
979  // Get corresponding basis vectors
980  eigen_sparse_t low_energy_basis = basisvectors * transformator;
981 
982  // --- Perform the Schrieffer Wolff transformation ---
983 
984  // Projector on the selected basis vectors (typically low-energy subspace)
985  eigen_sparse_t projector0 = low_energy_basis0 * low_energy_basis0.adjoint();
986  eigen_sparse_t projector = low_energy_basis * low_energy_basis.adjoint();
987 
988  // Reflection operator which change signes of the selectes basis vectors but act
989  // trivially on others
990  eigen_dense_t reflection0 =
991  eigen_dense_t::Identity(states.size(), states.size()) - 2 * projector0;
992  eigen_dense_t reflection =
993  eigen_dense_t::Identity(states.size(), states.size()) - 2 * projector;
994 
995  // Direct rotator from low_energy_basis to low_energy_basis0
996  eigen_sparse_t rotator = (reflection0 * reflection).sqrt().sparseView();
997 
998  if (std::isnan(std::abs(rotator.coeffRef(0, 0)))) {
999  throw std::runtime_error(
1000  "Error in calculating the matrix square root. Try to use the picomplex module or "
1001  "increase accuracy by choosing smaller thresholds.");
1002  }
1003 
1004  // Calculate the effective Hamiltonian
1005  transformator = basisvectors.adjoint() * rotator.adjoint() * projector0 * low_energy_basis0;
1006 
1007  this->applyRightsideTransformator(transformator);
1008  }
