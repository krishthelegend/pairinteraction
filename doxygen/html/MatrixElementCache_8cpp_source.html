49 bool selectionRulesMultipoleNew(StateOne const &state1, StateOne const &state2, int kappa, int q) {
50  bool validL = (abs(state1.getL() - state2.getL()) <= kappa) &&
51  (kappa % 2 == abs(state1.getL() - state2.getL()) % 2);
52  bool validJ =
53  (fabs(state1.getJ() - state2.getJ()) <= kappa) && (state1.getJ() + state2.getJ() >= kappa);
54  bool validM = state1.getM() == state2.getM() + q;
55  bool validQ = abs(q) <= kappa;
56  bool noZero = !(kappa == 2 && state1.getJ() == state2.getJ() && state2.getJ() == 1.5 &&
57  state1.getM() == -state2.getM() && fabs(state1.getM() - state2.getM()) == 1);
58  return validL && validJ && validM && validQ && noZero;
59 }

79 MatrixElementCache::MatrixElementCache(std::string const &cachedir)
80  : defectdbname(""), dbname((boost::filesystem::absolute(cachedir) /
81  ("cache_elements_" + version::cache() + ".db"))
82  .string()),
83  db(new sqlite::handle(dbname)), stmt(new sqlite::statement(*db)),
84  pid_which_created_db(utils::get_pid()) {
85 
86  // Speed up database access
87  stmt->exec("PRAGMA synchronous = OFF"); // do not wait on write, hand off to OS and carry on
88  stmt->exec(
89  "PRAGMA journal_mode = MEMORY"); // keep rollback journal in memory during transaction
90 
91  // Create cache tables (reduced_momentum_s and reduced_momentum_l need not to be cached since
92  // they are trivial)
93  stmt->exec("create table if not exists cache_radial ("
94  "method int, species text, k integer, n1 integer, l1 integer, j1 double,"
95  "n2 integer, l2 integer, j2 double, value double, primary key (method, species, k, "
96  "n1, l1, j1, n2, l2, j2)) without rowid;");
97 
98  stmt->exec(
99  "create table if not exists cache_angular ("
100  "k integer, j1 double, m1 double,"
101  "j2 double, m2 double, value double, primary key (k, j1, m1, j2, m2)) without rowid;");
102 
103  stmt->exec(
104  "create table if not exists cache_reduced_commutes_s ("
105  "s double, k integer, l1 integer, j1 double,"
106  "l2 integer, j2 double, value double, primary key (s, k, l1, j1, l2, j2)) without rowid;");
107 
108  stmt->exec(
109  "create table if not exists cache_reduced_commutes_l ("
110  "s double, k integer, l1 integer, j1 double,"
111  "l2 integer, j2 double, value double, primary key (s, k, l1, j1, l2, j2)) without rowid;");

131 void MatrixElementCache::loadElectricDipoleDB(std::string const &path, std::string const &species) {
132  int kappa_angular = 1;
133  float s = 0.5;
134  if (std::isdigit(species.back()) != 0) {
135  s = ((species.back() - '0') - 1) / 2.;
136  }
137 
138  // Get the radial matrix elements from the CSV file
139  boost::escaped_list_separator<char> els("", ";", "\"\'");
140  std::vector<std::pair<CacheKey_cache_radial, double>> radial_raw_data;
141  std::ifstream ifs(path.c_str());
142  std::string line;
143  bool firstline = true;
144  while (std::getline(ifs, line)) {
145  boost::tokenizer<boost::escaped_list_separator<char>> tk(line, els);
146  std::vector<std::string> tk_vec(tk.begin(), tk.end());
147 
148  try {
149  int n1 = std::stoi(tk_vec[0]);
150  int l1 = std::stoi(tk_vec[1]);
151  float j1 = std::stof(tk_vec[2]);
152  int n2 = std::stoi(tk_vec[3]);
153  int l2 = std::stoi(tk_vec[4]);
154  float j2 = std::stof(tk_vec[5]);
155  double val = std::stod(tk_vec[6]);
156 
157  // Convert the radial matrix element to um
158  val *= au2um;
159 
160  // Store the radial matrix element
161  auto key1 =
162  CacheKey_cache_radial(method, species, kappa_angular, n1, n2, l1, l2, j1, j2);
163  radial_raw_data.emplace_back(key1, val);
164 
165  // Store which expressions are needed for converting the radial matrix element to a
166  // reduced radial matrix element
167  auto key3 = CacheKey_cache_reduced_commutes(s, kappa_angular, l1, l2, j1, j2);
168  cache_reduced_commutes_s_missing.insert(key3);
169  auto key4 = CacheKey_cache_reduced_multipole(kappa_angular, l1, l2);
170  cache_reduced_multipole_missing.insert(key4);
171 
172  } catch (std::invalid_argument &e) {
173  if (!firstline) { // Skip header
174  std::cerr << "WARNING: During loading the electric dipole database, the error '"
175  << e.what() << "' occured." << std::endl;
176  }
177  }
178 
179  firstline = false;
180  }
181  ifs.close();
182 
183  // Precaculate the expressions
184  this->update();
185 
186  // Calculate the reduced radial matrix element and save it to the in-memory cache (it's not
187  // saved to the sqlite database)
188  for (auto const &pair : radial_raw_data) {
189  auto key1 = pair.first;
190  auto key3 = CacheKey_cache_reduced_commutes(s, kappa_angular, key1.l[0], key1.l[1],
191  key1.j[0], key1.j[1]);
192  auto key4 = CacheKey_cache_reduced_multipole(kappa_angular, key1.l[0], key1.l[1]);
193  double val = pair.second /
194  (key3.sgn * cache_reduced_commutes_s[key3] * key4.sgn * cache_reduced_multipole[key4]);
195 
196  cache_radial[key1] = val;
197  }
198 }

204 MatrixElementCache::CacheKey_cache_radial::CacheKey_cache_radial(method_t method,
205  std::string species, int kappa,
206  int n1, int n2, int l1, int l2,
207  float j1, float j2)
208  : species(std::move(species)), method(method), kappa(kappa) {
209  if ((n1 < n2) || ((n1 == n2) && ((l1 < l2) || ((l1 == l2) && (j1 <= j2))))) {
210  n = {{n1, n2}};
211  l = {{l1, l2}};
212  j = {{j1, j2}};
213  } else {
214  n = {{n2, n1}};
215  l = {{l2, l1}};
216  j = {{j2, j1}};
217  }
218 }

220 MatrixElementCache::CacheKey_cache_angular::CacheKey_cache_angular(int kappa, float j1, float j2,
221  float m1, float m2)
222  : kappa(kappa) {
223  if ((j1 < j2) || ((j1 == j2) && (m1 <= m2))) {
224  j = {{j1, j2}};
225  m = {{m1, m2}};
226  sgn = 1;
227  } else {
228  j = {{j2, j1}};
229  m = {{m2, m1}};
230  sgn = pow(-1, int(j1 - m1 + j2 - m2));
231  }
232 }

234 MatrixElementCache::CacheKey_cache_reduced_commutes::CacheKey_cache_reduced_commutes(
235  float s, int kappa, int l1, int l2, float j1, float j2)
236  : s(s), kappa(kappa) {
237  if ((l1 < l2) || ((l1 == l2) && (j1 <= j2))) {
238  l = {{l1, l2}};
239  j = {{j1, j2}};
240  sgn = 1;
241  } else {
242  l = {{l2, l1}};
243  j = {{j2, j1}};
244  sgn = pow(-1, int(l1 + j1 + l2 + j2 + 2 * s)); // TODO is this formula always correct?
245  }

358 double MatrixElementCache::getMagneticDipole(StateOne const &state_row, StateOne const &state_col) {
359  if (state_row.getSpecies() != state_col.getSpecies()) {
360  throw std::runtime_error("The species must be the same for the final and initial state.");
361  }
362 
363  const std::string &species = state_row.getSpecies();
364  const float &s = state_row.getS();
365 
366  // Search cache for constituents of the matrix element
367  auto key1 = CacheKey_cache_radial(method, species, 0, state_row.getN(), state_col.getN(),
368  state_row.getL(), state_col.getL(), state_row.getJ(),
369  state_col.getJ());
370  auto iter1 = cache_radial.find(key1);
371  if (iter1 == cache_radial.end()) {
372  cache_radial_missing.insert(key1);
373  }
374 
375  auto key2 = CacheKey_cache_angular(1, state_row.getJ(), state_col.getJ(), state_row.getM(),
376  state_col.getM());
377  auto iter2 = cache_angular.find(key2);
378  if (iter2 == cache_angular.end()) {
379  cache_angular_missing.insert(key2);
380  }
381 
382  auto key3 = CacheKey_cache_reduced_commutes(s, 1, state_row.getL(), state_col.getL(),
383  state_row.getJ(), state_col.getJ());
384  auto iter3 = cache_reduced_commutes_s.find(key3);
385  if (iter3 == cache_reduced_commutes_s.end()) {
386  cache_reduced_commutes_s_missing.insert(key3);
387  }
388 
389  auto key4 = CacheKey_cache_reduced_commutes(s, 1, state_row.getL(), state_col.getL(),
390  state_row.getJ(), state_col.getJ());
391  auto iter4 = cache_reduced_commutes_l.find(key4);
392  if (iter4 == cache_reduced_commutes_l.end()) {
393  cache_reduced_commutes_l_missing.insert(key4);
394  }
395 
396  // Update cache by calculate missing constituents
397  if (this->update() != 0) {
398  if (iter1 == cache_radial.end()) {
399  iter1 = cache_radial.find(key1);
400  }
401  if (iter2 == cache_angular.end()) {
402  iter2 = cache_angular.find(key2);
403  }
404  if (iter3 == cache_reduced_commutes_s.end()) {
405  iter3 = cache_reduced_commutes_s.find(key3);
406  }
407  if (iter4 == cache_reduced_commutes_l.end()) {
408  iter4 = cache_reduced_commutes_l.find(key4);
409  }
410  }
411 
412  return -bohr_magneton * iter1->second * key2.sgn * iter2->second *
413  (gL * key3.sgn * iter3->second *
414  sqrt(state_row.getL() * (state_row.getL() + 1) * (2 * state_row.getL() + 1)) +
415  gS * key4.sgn * iter4->second * sqrt(s * (s + 1) * (2 * s + 1)));
416 }

545  return std::pow(au2um, power) * IntegrateRadialElement<Numerov>(qd1, power, qd2);

547  if (method == WHITTAKER) {
548  return std::pow(au2um, power) * IntegrateRadialElement<Whittaker>(qd1, power, qd2);
549  }
550  std::string msg("You have to provide all radial matrix elements on your own because you have "
551  "deactivated the calculation of missing radial matrix elements!");
552  std::cout << msg << std::endl;
553  throw std::runtime_error(msg);

556 void MatrixElementCache::precalculate(const std::vector<StateOne> &basis_one, int kappa_angular,
557  int q, int kappa_radial, bool calcElectricMultipole,
558  bool calcMagneticMomentum, bool calcRadial) {
559 
560  std::string species;
561  float s = std::numeric_limits<float>::max();
562 
563  // --- Determine elements ---
564 
565  for (size_t idx_col = 0; idx_col < basis_one.size(); ++idx_col) {
566  const auto &state_col = basis_one[idx_col];
567  if (state_col.isArtificial()) {
568  continue;
569  }
570 
571  if (species.empty()) {
572  species = state_col.getSpecies();
573  s = state_col.getS();
574  }
575 
576  for (size_t idx_row = 0; idx_row <= idx_col; ++idx_row) {
577  const auto &state_row = basis_one[idx_row];
578  if (state_row.isArtificial()) {
579  continue;
580  }
581 
582  if (q != std::numeric_limits<int>::max() && state_row.getM() - state_col.getM() != q) {
583  continue;
584  }
585 
586  if ((calcElectricMultipole &&
587  selectionRulesMultipoleNew(state_row, state_col, kappa_angular)) ||
588  (calcMagneticMomentum && selectionRulesMomentumNew(state_row, state_col)) ||
589  calcRadial) {
590  if (calcElectricMultipole || calcMagneticMomentum || calcRadial) {
591  auto key = CacheKey_cache_radial(
592  method, species, kappa_radial, state_row.getN(), state_col.getN(),
593  state_row.getL(), state_col.getL(), state_row.getJ(), state_col.getJ());
594  if (cache_radial.find(key) == cache_radial.end()) {
595  cache_radial_missing.insert(key);
596  }
597  }
598 
599  if (calcElectricMultipole || calcMagneticMomentum) {
600  auto key =
601  CacheKey_cache_angular(kappa_angular, state_row.getJ(), state_col.getJ(),
602  state_row.getM(), state_col.getM());
603  if (cache_angular.find(key) == cache_angular.end()) {
604  cache_angular_missing.insert(key);
605  }
606  }
607 
608  if (calcElectricMultipole || calcMagneticMomentum) {
609  auto key = CacheKey_cache_reduced_commutes(s, kappa_angular, state_row.getL(),
610  state_col.getL(), state_row.getJ(),
611  state_col.getJ());
612  if (cache_reduced_commutes_s.find(key) == cache_reduced_commutes_s.end()) {
613  cache_reduced_commutes_s_missing.insert(key);
614  }
615  }
616 
617  if (calcMagneticMomentum) {
618  auto key = CacheKey_cache_reduced_commutes(s, kappa_angular, state_row.getL(),
619  state_col.getL(), state_row.getJ(),
620  state_col.getJ());
621  if (cache_reduced_commutes_l.find(key) == cache_reduced_commutes_l.end()) {
622  cache_reduced_commutes_l_missing.insert(key);
623  }
624  }
625 
626  if (calcElectricMultipole) {
627  auto key = CacheKey_cache_reduced_multipole(kappa_angular, state_row.getL(),
628  state_col.getL());
629  if (cache_reduced_multipole.find(key) == cache_reduced_multipole.end()) {
630  cache_reduced_multipole_missing.insert(key);
631  }
632  }
633  }
634  }
635  }
